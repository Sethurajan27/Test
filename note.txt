def get_token(self):
    payload = {
        'address': self.account['address'],
        'password': self.account['password']
    }
    response = self.session.post(f'{self.base_url}/token', json=payload, verify=False)

    # ✅ Debugging print
    print("Token API response:", response.text)

    try:
        return response.json()['token']
    except KeyError:
        raise ValueError("❌ Token not found in response. Likely signup or API error.")
