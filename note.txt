import openai
from datetime import datetime
import json

def load_prompts():
    with open("prompts/tamil_news_prompts.json", "r", encoding="utf-8") as f:
        return json.load(f)

def generate_news(api_key):
    openai.api_key = api_key
    today = datetime.now().strftime("%Y-%m-%d")
    prompts = load_prompts()
    texts = []

    for period in prompts:
        full_prompt = f"{prompts[period]['prompt'].replace('{DATE}', today)}"
        print(f"Generating for {period}...")
        response = openai.ChatCompletion.create(
            model="gpt-4",
            messages=[{"role": "user", "content": full_prompt}],
        )
        texts.append((period, response.choices[0].message["content"]))
    
    return texts

import requests
import time
import os

def generate_speech(api_key, text, label, voice_id='EXAVITQu4vr4xnSDxMaL'):
    url = f'https://api.elevenlabs.io/v1/text-to-speech/{voice_id}'
    headers = {
        'xi-api-key': api_key,
        'Content-Type': 'application/json',
        'Accept': 'audio/mpeg'
    }
    payload = {
        'text': text,
        'model_id': 'eleven_monolingual_v1',
        'voice_settings': {
            'stability': 0.2,
            'similarity_boost': 0.8,
            'style': 0.6,
            'use_speaker_boost': True
        }
    }
    response = requests.post(url, headers=headers, json=payload)
    filename = f"{label}_{int(time.time())}.mp3"
    with open(filename, 'wb') as f:
        f.write(response.content)
    return filename

Handles creation of temporary emails using the mail.tm API.

import requests
import time
import uuid

class MailTMClient:
    def __init__(self):
        self.base_url = "https://api.mail.tm"
        self.domain = self._get_domain()
        self.account = self._create_account()
        self.token = self._get_token()

    def _get_domain(self):
        response = requests.get(f"{self.base_url}/domains")
        response.raise_for_status()
        domains = response.json()["hydra:member"]
        return domains[0]["domain"]

    def _create_account(self):
        email = f"{uuid.uuid4().hex[:8]}@{self.domain}"
        password = uuid.uuid4().hex
        payload = {"address": email, "password": password}
        response = requests.post(f"{self.base_url}/accounts", json=payload)

        # Retry once if account creation fails
        if response.status_code != 201:
            raise Exception(f"Failed to create account: {response.text}")

        return {"email": email, "password": password}

    def _get_token(self):
        payload = {
            "address": self.account["email"],
            "password": self.account["password"]
        }
        response = requests.post(f"{self.base_url}/token", json=payload)
        response.raise_for_status()
        return response.json()["token"]

    def wait_for_email(self, subject_keyword, timeout=120):
        headers = {"Authorization": f"Bearer {self.token}"}
        start_time = time.time()

        while time.time() - start_time < timeout:
            response = requests.get(f"{self.base_url}/messages", headers=headers)
            response.raise_for_status()
            messages = response.json()["hydra:member"]

            for msg in messages:
                if subject_keyword.lower() in msg["subject"].lower():
                    msg_id = msg["id"]
                    msg_data = requests.get(f"{self.base_url}/messages/{msg_id}", headers=headers).json()
                    return msg_data["text"]

            time.sleep(5)

        raise TimeoutError("Email not received within timeout")

from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.chrome.options import Options
import time
import re

def elevenlabs_signup(driver, email, password):
    try:
        driver.get("https://www.elevenlabs.io/sign-up")
        time.sleep(3)

        email_input = driver.find_element(By.NAME, "email")
        email_input.send_keys(email)
        password_input = driver.find_element(By.NAME, "password")
        password_input.send_keys(password)

        driver.find_element(By.TAG_NAME, "button").click()
        time.sleep(10)  # Wait for dashboard to load

        driver.get("https://elevenlabs.io/app/account")
        time.sleep(5)

        page_text = driver.page_source
        match = re.search(r'"apiKey"\s*:\s*"(.+?)"', page_text)
        if match:
            return match.group(1)
        else:
            raise ValueError("API key not found on ElevenLabs account page.")

    except Exception as e:
        print(f"ElevenLabs signup failed: {e}")
        driver.quit()
        raise
