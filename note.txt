        google.script.run.withSuccessHandler(function(result) {
          document.getElementById("status").textContent = `Started: ${result.email}`;
        }).startMonitoring();
      }

        google.script.run.withSuccessHandler(function(result) {
          document.getElementById("status").textContent = result.status;
        }).stopMonitoring();
