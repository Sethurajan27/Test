name: News Login Test

on: [push, workflow_dispatch]

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      
      - name: Set up Python
        uses: actions/setup-python@v4
        with:
          python-version: '3.10'
      
      - name: Install Chrome
        run: |
          sudo apt-get update
          sudo apt-get install -y chromium-browser chromium-chromedriver
      
      - name: Install dependencies
        run: |
          pip install -r requirements.txt
      
      - name: Run news login test
        env:
          NEWS_URL: ${{ secrets.NEWS_URL }}
          NEWS_USERNAME: ${{ secrets.NEWS_USERNAME }}
          NEWS_PASSWORD: ${{ secrets.NEWS_PASSWORD }}
          NEWS_USERNAME_FIELD: ${{ secrets.NEWS_USERNAME_FIELD }}
          NEWS_PASSWORD_FIELD: ${{ secrets.NEWS_PASSWORD_FIELD }}
          NEWS_LOGIN_BUTTON: ${{ secrets.NEWS_LOGIN_BUTTON }}
          HEADLESS: "true"
        run: |
          python -m src.news_login
      
      - name: Upload screenshots
        if: always()
        uses: actions/upload-artifact@v3
        with:
          name: news-login-screenshots
          path: |
            news_login_success.png
            login_error.png
